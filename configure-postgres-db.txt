Configure Postgres Database for Local Machine
==============================================

Installing Postgres Server on Windows:
--------------------------------------
https://www.enterprisedb.com/downloads/postgres-postgresql-downloads

* When you run through the install wizard, you need to install the PostgreSQL Server.
When it asks for you to set a password, I use Ducksdatabase5! and keep the port as 5432.
Next, you need to find the bin path to PostgreSQL (e.g. “C:\Program Files\PostgreSQL\14\bin”).
Copy the full path to the bin folder. Then add it to your PATH...

* Settings > System > About > Advanced System Settings >
Environment Variables… > Double click the Path Variable > New > Paste in the
path > OK.

Testing the Database Connection, Create Tables

* After adding to your path, open a console and type the following command:

psql postgres://postgres@localhost

* Enter the password. After login, copy the following CREATE statements,
paste them into your console, and press the enter key:

CREATE TABLE "Users_Registered" (
	"User_ID"	SERIAL,
	"Reg_Timestamp"	BIGINT,
	PRIMARY KEY("User_ID")
);

CREATE TABLE "Users_Online" (
	"User_ID"	SERIAL,
	"Login_Timestamp"	BIGINT,
	PRIMARY KEY("User_ID"),
	FOREIGN KEY("User_ID") REFERENCES "Users_Registered"("User_ID") ON DELETE CASCADE
);

CREATE TABLE "Session" (
	"Session_ID"	SERIAL,
	"Region"	TEXT,
	"Country"	TEXT,
	"Visited_Timestamp"	BIGINT,
	PRIMARY KEY("Session_ID")
);

CREATE TABLE "Pages_Viewed" (
    	"Page_ID"    SERIAL,
    	"Page_Name"    TEXT,
    	"Viewed_Timestamp"    BIGINT,
    	"Session_ID"    INTEGER,
    	FOREIGN KEY("Session_ID") REFERENCES "Session"("Session_ID") ON DELETE CASCADE,
    	PRIMARY KEY("Page_ID")
);

CREATE TABLE "Pet_Sales" (
	"Pet_Sale_ID"	SERIAL,
	"Category"	TEXT,
	"Breed"	TEXT,
	"Date_Sold_Timestamp"	BIGINT,
	PRIMARY KEY("Pet_Sale_ID")
);

CREATE TABLE "Product_Sales" (
	"Product_Sale_ID"	SERIAL,
	"Description"	TEXT,
	"Quantity"	INTEGER,
	"Date_Sold_Timestamp"	BIGINT,
	PRIMARY KEY("Product_Sale_ID")
);
